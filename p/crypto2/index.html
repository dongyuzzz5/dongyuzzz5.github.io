def continued_fraction(e, n):
    """è®¡ç®—e/nçš„è¿åˆ†æ•°å±•å¼€"""
    cf = []
    while n:
        q = e // n
        cf.append(q)
        e, n = n, e % n
    return cf

def gradual_fractions(cf):
    """è®¡ç®—è¿åˆ†æ•°çš„æ¸è¿›åˆ†æ•°"""
    numerators = [0, 1]
    denominators = [1, 0]
    for i, q in enumerate(cf):
        numerators.append(q * numerators[i+1] + numerators[i])
        denominators.append(q * denominators[i+1] + denominators[i])
        yield numerators[-1], denominators[-1]

def wiener_attack(e, n):
    """æ‰§è¡ŒWieneræ”»å‡»"""
    cf = continued_fraction(e, n)
    for k, d in gradual_fractions(cf):
        if k == 0 or d == 0:
            continue
        # æ£€æŸ¥æ˜¯å¦æ»¡è¶³ed â‰¡ 1 mod k
        if (e * d - 1) % k != 0:
            continue
        phi = (e * d - 1) // k
        # è§£æ–¹ç¨‹x^2 - (n - phi + 1)x + n = 0
        b = n - phi + 1
        discriminant = b * b - 4 * n
        if discriminant < 0:
            continue
        sqrt_discriminant = math.isqrt(discriminant)
        if sqrt_discriminant * sqrt_discriminant != discriminant:
            continue
        p = (b + sqrt_discriminant) // 2
        q = (b - sqrt_discriminant) // 2
        if p * q == n:
            return d
